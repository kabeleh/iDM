\providecommand{\HeatmapInclude}[1]{%
    \ifcsname ifLINKBOXES\endcsname
        \ifLINKBOXES
            \resizebox{\linewidth}{!}{\input{#1.pgf}}%
        \else
            \includegraphics[width=\linewidth]{#1.pdf}%
        \fi
    \else
        \resizebox{\linewidth}{!}{\input{#1.pgf}}%
    \fi
}

\providecommand{\SWGCHeatmapCaptionLead}{%
    Posterior heatmaps for the \gls{sswgc} diagnostic of the \Nref{pot:tanh} potential.%
}

\providecommand{\SWGCHeatmapCaptionNote}{%
    The \gls{sswgc} demands that scalar field interactions
    (second panel from the top, $2\left(V^{(3)}\right)^2-V^{\prime\prime}V^{(4)}$)
    are stronger than gravitational interactions
    (top panel, $\left(V^{\prime\prime}/M_\textnormal{P}\right)^2$).
    The \gls{sswgc} is violated if their difference (third panel) is negative.
    The panels show the posterior distribution of our sample, with point-wise bins in
    redshift $z$ as a heatmap, the posterior mean as a dashed line, and the global
    best-fit parameter evolution as a solid line.
    The bottom panel shows the probability of \gls{sswgc} violations in our sample
    of 5000 trajectories of accepted \gls{mcmc} steps.
    The \gls{sswgc} uses the difference of three terms that are products of numerically
    small terms. The uncertainty due to numerical noise is indicated in the bottom panel
    as a shaded area around the red probability curve.
    The bottom panel also shows the number count at each redshift step of different
    trajectories among our sample that violate the \gls{sswgc}.%
}
